52void SASPSO<dim>::initialize_parallel()
53{
54 // Create a shared pointer to the problem
55 std::shared_ptr<Problem<dim>> problem = std::make_shared<Problem<dim>>(Optimizer<dim>::problem_);
56
57 // Initialize the array of total constraint violations. Needed since threads performs direct access to the array
58 std::vector<double> total_violations;
59 total_violations.resize(swarm_size_);
60
61 // Initialize the swarm. Needed since threads performs direct access to the array
62 swarm_.resize(swarm_size_);
63
64 // Add the constraint violation to the array
65 total_violations[0] = swarm_[0].get_best_constraint_violation();
66
67 // Initialize the global best
68 global_best_index_ = 0;
69
70// Spawn the threads to initialize the particles
71#pragma omp parallel shared(swarm_, total_violations, global_best_index_, problem)
72 {
73 // Instantiate the marsenne twister, one for each thread since it is not thread safe
74 std::random_device rand_dev;
75 std::shared_ptr<std::mt19937> generator = std::make_shared<std::mt19937>(rand_dev());
76
77 // Initialize all the particles. Barrier at the end of the parallel region is needed
78#pragma omp for schedule(static)
79 for (std::size_t i = 0; i < swarm_size_; ++i)
80 {
81 // Create and initialize the particle
82 swarm_[i] = Particle_SASPSO(problem, generator, omega_s_, omega_f_, phi1_s_, phi1_f_, phi2_s_, phi2_f_);
83 swarm_[i].initialize();
84 // Add the constraint violation to the array
85 total_violations[i] = swarm_[i].get_best_constraint_violation();
86 }
87#pragma omp single
88 {
89 // Initialize the violation threshold as the median of the total violations
90 std::sort(total_violations.begin(), total_violations.end());
91 if (swarm_size_ % 2 == 0)
92 violation_threshold_ = (total_violations[swarm_size_ / 2] + total_violations[swarm_size_ / 2 - 1]) / 2;
93 else
94 violation_threshold_ = total_violations[swarm_size_ / 2];
95 }
96
97 // Initialize the thread local best index
98 std::size_t local_best_index = global_best_index_;
99
100 // Find the local best index for each thread
101#pragma omp for nowait schedule(static)
102 for (std::size_t i = 1; i < swarm_size_; ++i)
103 {
104 if (swarm_[i].is_better_than(swarm_[local_best_index], violation_threshold_, tol_))
105 local_best_index = i;
106 }
107
108 // Reduction: Each thread updates the global best index if it has a better solution
109#pragma omp critical(update_global_best_init)
110 {
111 if (swarm_[local_best_index].is_better_than(swarm_[global_best_index_], violation_threshold_, tol_))
112 global_best_index_ = local_best_index;

117template <std::size_t dim>
118void SASPSO<dim>::optimize(std::ostream& history_out, std::ostream& simulation_out, const int interval)
119{
120 int current_iter = 0;
121
122 // Outer optimization loop over all the iterations
123 while (current_iter < max_iter_)
124 {
125 // Reset the number of feasible particles for the current iteration
126 int feasible_particles = 0;
127
128 // Process each particle of the swarm
129 for (size_t i = 0; i < swarm_size_; ++i)
130 {
131 // Update the particle
132 swarm_[i].update(swarm_[global_best_index_].get_best_position(), violation_threshold_, current_iter, max_iter_, tol_);
133 // Check if the particle is feasible
134 if (swarm_[i].get_best_constraint_violation() <= violation_threshold_)
135 feasible_particles++;
136 }
137
138 // Update the violation threshold according to the proportion of feasible particles
139 violation_threshold_ = violation_threshold_ * (1.0 - (feasible_particles / (double)swarm_size_));
140 violation_threshold_ = violation_threshold_ < tol_ ? 0 : violation_threshold_;
141
142 // Update the global best position
143 for (std::size_t i = 0; i < swarm_size_; ++i)
144 if (swarm_[i].is_better_than(swarm_[global_best_index_], violation_threshold_, tol_))
145 global_best_index_ = i;
146
147 // Log the optimization process in both history and simulation streams.
148 // NB: Branch prediction will optimize well the following branch since it has a constant condition
149 if (Optimizer<dim>::log_verbose_ && current_iter % interval == 0)
150 {
151 // Print Iters,value,violation,threshold,feasible_particles to the history
152 history_out << current_iter << ","
153 << swarm_[global_best_index_].get_best_value() << ","
154 << swarm_[global_best_index_].get_best_constraint_violation() << ","
155 << violation_threshold_ << ","
156 << feasible_particles << std::endl;
157 // Print all the particles position and flag the best one to the simulation stream
158 for (int i = 0; i < swarm_size_; i++)
159 {
160 simulation_out << current_iter << ",";
161 for (int d = 0; d < dim; d++)
162 simulation_out << swarm_[i].get_best_position()[d] << ",";
163 simulation_out << ((i == global_best_index_) ? 1 : 0) << std::endl;
164 }
165 }
166
167 // Update the current iteration
168 current_iter++;
169 }
170}
