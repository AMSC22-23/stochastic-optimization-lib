115void Bee<dim>::compute_probability(const double total_fitness_value, const double total_constraint_violation, const double tol)
116{
117 // If the solution is feasible, the probability is proportional to the fitness value

131double Bee<dim>::compute_constraint_violation(const RealVector<dim> &position) const
132{

145template <size_t dim>
146bool Bee<dim>::feasibility_rule(const double value, const double viol, const double tol) const
147{
148 // (a) a feasible solution is preferred over an infeasible solution
149 if (this->constraint_violation_ < tol && viol > tol)
150 return true;
151 else if (this->constraint_violation_ > tol && viol < tol)
152 return false;
153 // (b) among two feasible solutions, the one with better objective function value is preferred
154 else if (this->constraint_violation_ < tol && viol < tol)
155 return this->cost_value_ < value;
156 // (c) among two infeasible solutions, the one with smaller TAV is chosen
157 else
158 return this->constraint_violation_ < viol;
159}
