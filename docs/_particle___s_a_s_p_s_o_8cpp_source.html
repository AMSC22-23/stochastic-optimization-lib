29void Particle_SASPSO<dim>::update(const RealVector<dim> &global_best_position, double violation_threshold, int iteration, int max_iter, double tol)
30{
31 // Compute beta avoiding division by zero
32 double beta = (global_best_position - best_position_).norm();
33 if (beta < 1e-16)
34 beta = 1e-16;
35 // Compute omega according to the current iteration
36 double delta = (omega_s_ - omega_f_) / max_iter;
37 double omega = (omega_s_ - omega_f_) * exp(-delta * iteration / beta) + omega_f_;
38 // Compute phi1 and phi2 according to the current iteration
39 delta = (phi1_s_ - phi1_f_) / max_iter;
40 double phi1 = (phi1_s_ - phi1_f_) * exp(-delta * iteration / beta) + phi1_f_;
41 // Compute phi2 according to the current iteration
42 delta = (phi2_s_ - phi2_f_) / max_iter;
43 double phi2 = (phi2_s_ - phi2_f_) * exp(delta * iteration / beta) + phi2_f_;
44
45 // Compute the isobarycenter of the particle
46 RealVector<dim> G = this->position_;
47 G += (phi1 * (best_position_ - this->position_) + phi2 * (global_best_position - this->position_)) / 3.0;
48
49 // Generate a point in the hypersphere of radius ||G-position|| centered in G
50 double radius = (G - this->position_).norm();
51 std::uniform_real_distribution<double> unif_distr(0, 1);
52 std::normal_distribution<double> normal_distr(0, 1);
53 // 1. Generate a random direction in a unitary hypersphere
54 RealVector<dim> random_direction = RealVector<dim>::NullaryExpr([&](int)
55 { return normal_distr(*this->random_generator_); });
56 random_direction.normalize();
57 // 2. Generate a random radius with probability proportional to the surface area of a ball with a given radius
58 double random_radius = pow(unif_distr(*this->random_generator_), 1.0 / 2);
59 // 3. Generate the point in the affine hypersphere
60 RealVector<dim> random_point = G + radius * (random_direction * random_radius);
61
62 // Update the velocity
63 velocity_ = omega * velocity_ + random_point - this->position_;
64 // Update the position
65 this->position_ = velocity_ + this->position_;
66
67 // Modify each dimension of the position vector with its saturation energy
68 for (size_t i = 0; i < dim; i++)
69 {
70 if (this->position_[i] < this->problem_->get_lower_bound(i))
71 this->position_[i] = this->problem_->get_lower_bound(i);
72 else if (this->position_[i] > this->problem_->get_upper_bound(i))
73 this->position_[i] = this->problem_->get_upper_bound(i);
74 }

76 // Update the total constraint violation in the actual position

134template <size_t dim>
135bool Particle_SASPSO<dim>::feasibility_rule(double value1, double value2, double viol1, double viol2, double violation_threshold, double tol) const
136{
137 double ub = violation_threshold + tol;
138 double lb = std::max(violation_threshold - tol, 0.0);
139
140 // (a) a feasible solution is preferred over an infeasible solution
141 if (viol1 <= lb && viol2 > ub)
142 return true;
143 else if (viol1 > ub && viol2 <= lb)
144 return false;
145 // (b) among two feasible solutions, the one with better objective function value is preferred
146 else if (viol1 <= lb && viol2 <= lb)
147 return value1 < value2;
148 // (c) among two infeasible solutions, the one with smaller TAV is chosen
149 else
150 return viol1 < viol2;
151}
